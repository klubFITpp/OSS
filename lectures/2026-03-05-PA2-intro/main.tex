\section{PA2 naruby}

\begin{frame}{Co je jazyk C?}


\begin{itemize}
  \item Paradigmata
    \begin{itemize}
      \item Funkcionální (hlavním stavebním blokem C jsou funkce)
      \item Imperativní (popisujeme jak se co dělá, ne čeho chceme docílit)
      \item Procedurální (program rozdělujeme do procedur - funkcí - které se spouští v pořadí)\cite{geeks:procedural_language}
    \end{itemize}
  \item Relativně jednoduchý jazyk
    \begin{itemize}
      \item Hot take: vše je číslo, pole čísel, nebo n-tice čísel a polí
      \item Máme pár aritmetických / logických operací
      \item if/for/while na ovládání běhu
      \item Vše ostatní vychází z fungování OS/paměti (schováno ve funkcích)
    \end{itemize}
\end{itemize}

\end{frame}

\begin{frame}{Co tedy C++}

\begin{itemize}
  \item Prakticky rozšíření C (většinově funkční C je funkční C++)
  \item Drasticky komplikovanější (třídy, templates, exceptions, přetěžování, reference)
  \item C++ má vlastní ISO standard\cite{isocpp:n4860} - skoro 2000 stránek
  \item Paradigma Objektově Orientované Programování (OOP)
\end{itemize}

\end{frame}

\section{OOP}

\begin{frame}{Definujme si problém}

\begin{itemize}
  \item Řekněme, že tvoříme hru, potřebujeme uchovávat hráče
  \item Má počet životů (0 - 3), damage: 0 - inf, stav:
    \begin{itemize}
      \item Jumping
      \item Swimming
      \item OnGround
    \end{itemize}
\end{itemize}

\end{frame}

\begin{frame}[fragile]{Řešení v C}

\begin{minted}[fontsize=\small]{C}

int create_player(player_t *player) {
  player.name = malloc(...);
  player.health = 3;
  player.damage = 5;
  player.state = ON_GROUND;

  return 0;
}

void delete_player(player_t *player) {
  free(player.name);
  player.name = NULL;
}

\end{minted}

\end{frame}


\begin{frame}[fragile]{Řešení v C}

\begin{minted}[fontsize=\small]{C}

int main(int argc, char **argv) {
  player_t player;
  if (!create_player(&player)) {
    fprintf(stderr, "Could not create player...");
    return -1;
  }

  // Kdokoliv může rozbít náš předpoklad
  player.health = UINT_MAX;

  // Zároveň musíme volat delete před každým return
  delete_player(&player);
  return 0;
}

\end{minted}

\end{frame}

\begin{frame}[fragile]{Použití tříd}

\begin{minted}{Cpp}

int main() {
  Player p = Player::create();

  // 'health' is a private member of 'Player' (clang)
  p.health = 0;

  p.del();
  return 0;
}

\end{minted}

\end{frame}

\begin{frame}[fragile]{Další funkce}

\begin{minted}[fontsize=\small]{Cpp}

class Player {
  // ...
  Player &heal(unsigned int amount) {
    unsigned int newhealth = this->health + amount;
    if (newhealth > 3) {
      return;
    }
    if (newhealth == 0) {
      printf("Player died!\n");
    }

    this->health = newhealth;
    return *this;
  }
}
\end{minted}

\end{frame}

\begin{frame}{Čeho jsme dosáhli?}

\begin{itemize}
  \item Zapouzdřili jsme data o hráči do striktního rozhraní
    \begin{itemize}
      \item Omezili jsme, co za informace můžeme číst a jaké zapisovat
    \end{itemize}
  \item Abstrahovali jsme rozhraní pro vývojáře od toho, jaké datové typy používáme
    \begin{itemize}
      \item Jednoduše můžeme změnit health na např. \mintinline{Cpp}{char *}
    \end{itemize}
  \item Zajistili jsme si, že platí náš invariant (naše omezení životů atd.)
\end{itemize}

\end{frame}

\begin{frame}{RAII}

\begin{itemize}
  \item Resource Acquisition Is Initialization\footnote{Toto umí i C$^*$\cite{youtube:destructor_c}}\cite{cppreference:raii}
  \item Kompilátor umí \enquote{mazat} lokální proměnné
  \item Proč by neuměl místo mazání zavolat \texttt{del} (případně až po tom smazat)
  \item To existuje - destruktor, je to funkce pojmenovaná \texttt{~Type} (\texttt{~Player})
  \item Destruktor by neměl způsobovat chyby
\end{itemize}

\end{frame}

\begin{frame}[fragile]{RAII}

\begin{minted}{Cpp}

class Player {
  ...
  ~Player() {
    delete this->name;
    this->name = nullptr;
  }
}

\end{minted}

\end{frame}

\begin{frame}{Co jsou tedy ty konstruktory?}

\begin{itemize}
  \item Funkce, chovající se jako náš \texttt{create\_player}
    \begin{itemize}
      \item Je static = jejím \enquote{skrytým} argumentem není \texttt{Player}
      \item Buď dojde k chybě, nebo vytvoří (a vrátí) náš objekt
      \item Může mít víc variant, podle počtu argumentů (a jejich typů)
    \end{itemize}
  \item Pokud existuje konstruktor, je to jediný způsob, jak zvenčí vytvořit objekt
\end{itemize}

\end{frame}
