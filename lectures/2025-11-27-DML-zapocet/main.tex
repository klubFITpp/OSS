\begin{frame}

\begin{definition}[Fibonacciho posloupnost]
  $f_0 := 0$ \\ 
  $f_1 := 1$ \\
  $f_n := f_{n - 1} + f_{n - 2}$
\end{definition}

\begin{problem}
  Dokažte, že pro Fibonacciho posloupnost platí $f_n > \alpha^{n - 2}$ pro $\forall n \geq 3, \alpha = \frac{1 + \sqrt{5}}{2}$

  Z DML skript\cite{courses:DMLfib}
\end{problem}

\end{frame}

\begin{frame}

Triviálně ověříme, že tvrzení platí pro $n = 3$, $n = 4$:

\[
  f_3 = 2 > \alpha^1 = \frac{1 + \sqrt{5}}{2} \doteq 1.618
\]
\[
  f_4 = 3 > \alpha^2 = \frac{(1 + \sqrt{5})^2}{2^2} = \frac{1 + 2\sqrt{5} + 5}{4} = 1 + \frac{2 + 2\sqrt{5}}{4} = 1 + \frac{2 + \sqrt{5}}{2}
\]

\hspace{1em}

Nyní se snažíme dokázat, že za předpokladu, že pokud pro $n > 4$ platí $f_n > \alpha^{n - 2}$ a $f_{n - 1} > \alpha^{n - 3}$, platí i $f_{n + 1} > \alpha^{n - 1}$.

\[
  f_{n + 1} = f_n + f_{n - 1} \overset{IP}{>} \alpha^{n - 2} + \alpha^{n - 3} = \alpha^{n - 3}(\alpha + 1) = \alpha^{n - 3}(\alpha^2) = \alpha^{n - 1}
\]

Tedy vidíme, že $f_{n + 1} > \alpha^{n - 1}$ \qed

\end{frame}

\section{Množiny a zobrazení}

\begin{frame}{Množina vs Uspořádaná n-tice}

\begin{itemize}
  \item Ve množinách neřešíme pořadí prvků \begin{itemize}
      \item $\{A, B, C\} = \{C, A, B\}$
      \item Vnímáme operace jako "obsahuje prvek", průnik, sjednocení, doplněk...
    \end{itemize}
  \item Ve uspořádaných n-ticích pořadí řešíme \begin{itemize}
      \item $(A, B, C) \neq (C, A, B)$
      \item Vnímáme jako spoustu věcí (BI-AAG, BI-AG1), spíš než s celkem pracujeme s jednotlivými prvky.
      \item BI-LA1: Pokud vnímáme n-tici jako vektor, operace škálování (násobení) skalárem
    \end{itemize}
\end{itemize}

\end{frame}

\begin{frame}{Množinové vztahy}

\begin{itemize}
  \item Podmnožina $A \subseteq B$ \begin{itemize}
    \item<2-> vše co je v A je i v B
    \item<2-> $(\forall x \in \mathit{U})(x \in A \Rightarrow x \in B)$
  \end{itemize}
  \item Rovnost $A = B$ \begin{itemize}
    \item<3-> vše co je v A je v B a vše co je v B je v a
    \item<3-> $A \subseteq B \land B \subseteq A$
  \end{itemize}
  \item Vlastní podmnožina $A \varsubsetneq B$ \begin{itemize}
    \item<4-> A je podmnožina B ale není rovna
    \item<4-> $A \subseteq B \land A \neq B$
  \end{itemize}
  \item Potenční množina $\mathit{P}(A)$ \begin{itemize}
    \item<5-> Množina všech podmnožin A
    \item<5-> $\{X \subseteq \mathit{U}| X \subseteq A\}$
    \item<5-> POZOR! je to \textbf{množina množin}
  \end{itemize}
  \item \small{Definice ze skript\cite{courses:DMLsets}}
\end{itemize}

\end{frame}

\begin{frame}{Množinové operace}

\begin{itemize}
  \item Množinový doplněk $\overline{A}$ (v univerzu $\mathit{U}$) \begin{itemize}
      \item<2-> $\{x \in \mathit{U}; x \notin A\}$
    \end{itemize}
  \item Průnik množin $A \cap B$ \begin{itemize}
      \item<3-> $\{x \in \mathit{U}; x \in A \land x \in B\}$
    \end{itemize}
  \item Sjednocení množin $A \cup B$ \begin{itemize}
      \item<4-> $\{x \in \mathit{U}; x \in A \lor x \in B\}$
    \end{itemize}
  \item Rozdíl množin $A \backslash B$ \begin{itemize}
      \item<5-> $\{x \in \mathit{U}; x \in A \land x \notin B\}$
    \end{itemize}
  \item Kartézský součin $A \times B$ \begin{itemize}
      \item<6-> $\{(a, b); a \in A \land b \in B\}$
      \item<7-> POZOR! je to \textbf{množina uspořádaných dvojic}
    \end{itemize}
\end{itemize}

\end{frame}

\begin{frame}{Množina množin uspořádaných dvojic množin čísel}

\begin{itemize}
  \item Množiny a n-tice mohou být vnořené kolikrát chceme
  \item $\{\{(\emptyset, \{42\})\}, \{(\{1,2 3\}, \{4, 5\})\}\}$
  \item $\emptyset = \{\}, \emptyset \notin \{\}, \emptyset \in \{\emptyset\}$
\end{itemize}

\end{frame}

\begin{frame}{Zobrazení}

\begin{itemize}
  \item Znáte \begin{itemize}
    \item funkce $f(x) = 3x + 5$ ($\mathbb{R} \mapsto \mathbb{R}$)
    \item posloupnosti $a_0 = 0, a_n = a_{n - 1} + 5$ ($\mathbb{N} \mapsto \mathbb{R}$)
  \end{itemize}
  \item Přiřazujeme ke každému prvku \textbf{$D_f$} jeden prvek \textbf{$H_f$}
  \item $f: A \mapsto B$ \\
    $f(x) = y$, $f(5) = 1$
\end{itemize}

\end{frame}

\begin{frame}{Speciální zobrazení}

\begin{itemize}
  \item Injektivní (prosté) \begin{itemize}
    \item<2-> $(\forall x,y \in A)(f(x) = f(y) \Rightarrow x = y)$
    \item<2-> $(\forall x,y \in A)(x \neq y\Rightarrow f(x) \neq f(y) )$
  \end{itemize}
  \item Surjektivní (na) \begin{itemize}
    \item<3-> $(\forall y \in B)(\exists x \in A)(y = f(x))$
  \end{itemize}
  \item Bijektivní (bijekce) \begin{itemize}
    \item<4-> zobrazení je zároveň injektivní a surjektivní
  \end{itemize}

\end{itemize}

\end{frame}

\begin{frame}{Mohutnost množiny}

\begin{itemize}
  \item "počet prvků", ale obecnější
  \item Pokud existuje $f: A \mapsto B$ je injektivní, pak $|A| \leq |B|$
  \item Pokud existuje $f: A \mapsto B$ je bijekce, pak $|A| = |B|$
  \item Toto pak platí obecně, i pro spočitatelné / nespočitatelné nekonečno
\end{itemize}

\end{frame}
